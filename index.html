$(document).ready(function() { // Set the countdown time to 1 minute (60 seconds) var countdownTime = 60; // Update the countdown timer every second var countdownInterval = setInterval(updateCountdown, 1000); function updateCountdown() { // Format the countdown time as mm:ss var minutes = Math.floor(countdownTime / 60); var seconds = countdownTime % 60; var formattedTime = ("0" + minutes).slice(-2) + ":" + ("0" + seconds).slice(-2); // Display the formatted countdown time $(".timer").text(formattedTime); // Decrease the countdown time by 1 second countdownTime--; // When the countdown reaches 0, show the message if (countdownTime < 0) { clearInterval(countdownInterval); $(".timer").hide(); $(".message").show(); } } });
